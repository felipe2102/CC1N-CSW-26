CREATE TABLE cursos(
	codigo serial,
	duracao int NOT NULL,
	nome varchar(100) NOT NULL,
	preco int NOT NULL,
	cod_professor int NOT NULL,
	CONSTRAINT codigo_curso
		PRIMARY KEY (codigo),
	CONSTRAINT cod_professor FOREIGN KEY (cod_professor)
		REFERENCES professores(codigo)
)

CREATE TABLE matriculas(
	cod_curso int NOT NULL,
	cod_aluno int NOT NULL,
	numero serial,
	data_inicio date NOT NULL,
	CONSTRAINT cod_curso FOREIGN KEY (cod_curso)
		REFERENCES cursos(cod),
	CONSTRAINT cod_aluno FOREIGN KEY (cod_aluno)
		REFERENCES alunos (cadastro),
	CONSTRAINT n_matricula
		PRIMARY KEY (numero)
)

CREATE TABLE professores(
	codigo serial,
	nome varchar(100) NOT NULL,
	cod_endereco int NOT NULL,
	CONSTRAINT codigo_professor
		PRIMARY KEY (codigo)
	CONSTRAINT cod_endereco FOREIGN KEY (cod_endereco)
		REFERENCES endereco(codigo)
)

CREATE TABLE alunos(
	cadastro serial,
	nome varchar(100) NOT NULL,
	CONSTRAINT codigo_aluno
		PRIMARY KEY (cadastro)
)

CREATE TABLE endereco(
	codigo serial,
	rua varchar(100) NOT NULL,
	numero int NOT NULL,
	complemento varchar(100) NOT NULL,
	bairro varchar(100) NOT NULL,
	cidade varchar(100)NOT NULL,
	estado varchar(100) NOT NULL,
	cep char(8) NOT NULL,
	CONSTRAINT codigo_endereco
		PRIMARY KEY (codigo)
)

CREATE TABLE alunos_enderecos (
	cod_aluno int NOT NULL,
	cod_endereco int NOT NULL,
	CONSTRAINT cod_aluno FOREIGN KEY (cod_aluno)
		REFERENCES aluno(cadastro)
	CONSTRAINT cod_endereco FOREIGN KEY (cod_endereco)
		REFERENCES endereco(codigo)
)

CREATE TABLE alunos_telefones (
	cod_aluno int NOT NULL,
	cod_telefone int NOT NULL,
	CONSTRAINT cod_aluno FOREIGN KEY (cod_aluno)
		REFERENCES aluno(cadastro)
	CONSTRAINT cod_telefone FOREIGN KEY (cod_telefone)
		REFERENCES telefones(codigo)
)

CREATE TABLE professores_telefones(
	cod_professor int NOT NULL,
	cod_telefone int NOT NULL,
	CONSTRAINT cod_professor FOREIGN KEY (cod_professor)
		REFERENCES professores(codigo)

		REFERENCES telefones(codigo)
